134 void loop() {
135 
136  update_limits(); // Werte der Potis am ADC einlesen
137 
138  /********************************( Änderung am Taster )*****************************************************/
139  aktueller_taster = digitalRead(TASTER_DISPLAY); // Variable
140  if (( voriger_taster == HIGH ) and ( aktueller_taster == LOW)) { // wurde der Tasterstatus geändert
141  t0_display = millis() - dt_display_ms; // Wenn Taster gedrückt, gleich die Displayausgabe aktualisieren
142  displ_nr++; // Displaynummer eins hochzählen
143 
144  if ( displ_nr > 1 ) { // Wenn Displaynummer größer Eins...
145  displ_nr = 0; // ... die Displaynummer Null setzen
146  }
147 
148  }
149  voriger_taster = aktueller_taster;
150 
151  /********************************( Display Ausgabe )*****************************************************/
152  if ( ( millis() - t0_display ) > dt_display_ms ) { // wenn Displayablaufzeit vergangen, dann...
153  t0_display = millis(); // neuer Wert Displayzeit
154  update_lcd( // Displaywerte aktualisieren
155  displ_nr, // Variablen, die übergeben werden, hier angeben/eintragen
156  temp_luft_C,
157  hum_luft,
158  hum_erde_1_adc,
159  hum_erde_2_adc,
160  limit_feuchte_1_adc,
161  limit_feuchte_2_adc,
162  limit_temp_adc,
163  limit_luefter_adc
164  );
165  }
166 
167  /******************************** Messen Temperatur und Feuchtigkeit ***********************************/
168  if ( ( millis() - t0_temperaturmessung ) > dt_temperaturmessung_ms ) { // wenn Zeit für Temp-Messung abgelaufen, dann
169  t0_temperaturmessung = millis(); // Timer auf neuen Wert setzen
170  update_messwerte(); // Messwerte aktualisieren
171 
172  }
173 
174 
175  /******************************** Heizung <--> Temperatursensor DHT22 ***********************************/
176  if ( ( millis() - t0_heizung ) > dt_heizung_ms ) { // wenn Zeit für Heizen abgelaufen, dann
177  t0_heizung = millis(); // Timer auf neuen Wert setzen
178  if (temp_luft_C < limit_temp_adc ) { // wenn Lufttemperatur kälter als eingesteller Wert, dann
179  digitalWrite(RELAIS_HEIZUNG, LOW ); // heizen
180  }
181  else { // sonst
182  digitalWrite(RELAIS_HEIZUNG, HIGH ); // nicht heízen
183  }
184  }
185 
186  /******************************** Lüfter <--> Temperatursensor DHT22 ************************************/
187  if ( ( millis() - t0_luefter ) > dt_luefter_ms ) {
188  t0_luefter = millis();
189  if (temp_luft_C > limit_luefter_adc ) { // wenn Lufttemperatur wärmer als eingesteller Wert, dann
190  digitalWrite(RELAIS_LUEFTER, LOW ); // Lüfter einschalten (LOW = Relais EIN)
191  }
192  else { // sonst
193  digitalWrite(RELAIS_LUEFTER, HIGH ); // Lüfter ausschalten (High = Relais AUS)
194  }
195  }
196 
197  /******************************** LED-Licht <--> Helligkeitssensor *************************************/
198  if ( ( millis() - t0_helligkeit ) > dt_helligkeit_ms ) {
199  t0_helligkeit = millis();
200  if ( digitalRead( HELLIGKEITSSENSOR ) == HIGH ) { // wenn dunkel, dann:
201  digitalWrite(RELAIS_UV_LED, LOW ); // Licht anschalten (LOW = Relais EIN)
202  }
203  else { // sonst:
204  digitalWrite(RELAIS_UV_LED, HIGH ); // Licht ausschalten (High = Relais AUS)
205  }
206  }

213
