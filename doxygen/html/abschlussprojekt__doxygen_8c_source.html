348 void loop() {
349 
350  update_limits(); // Werte der Potis am ADC einlesen
351 
352  /********************************( Änderung am Taster )*****************************************************/
353  aktueller_taster = digitalRead(TASTER_DISPLAY); // Variable
354  if (( voriger_taster == LOW ) and ( aktueller_taster == HIGH)) { // wurde der Tasterstatus geändert
355  t0_display = millis() - dt_display_ms; // Wenn Taster gedrückt, gleich die Displayausgabe aktualisieren
356  displ_nr++; // Displaynummer eins hochzählen
357 
358  if ( displ_nr > 1 ) { // Wenn Displaynummer größer Eins...
359  displ_nr = 0; // ... die Displaynummer Null setzen
360  }
361 
362  }
363  voriger_taster = aktueller_taster;
364 
365  /********************************( Display Ausgabe )*****************************************************/
366  if ( ( millis() - t0_display ) > dt_display_ms ) { // wenn Displayablaufzeit vergangen, dann...
367  t0_display = millis(); // neuer Wert Displayzeit
368  update_lcd( // Displaywerte aktualisieren
369  displ_nr, // Variablen, die übergeben werden, hier angeben/eintragen
370  temp_luft_C,
371  hum_luft,
372  hum_erde_1_adc,
373  hum_erde_2_adc,
374  limit_feuchte_1_adc,
375  limit_feuchte_2_adc,
376  limit_temp_adc,
377  limit_luefter_adc
378  );
379  }
380 
381  /******************************** Messen Temperatur und Feuchtigkeit ***********************************/
382  if ( ( millis() - t0_temperaturmessung ) > dt_temperaturmessung_ms ) { // wenn Zeit für Temp-Messung abgelaufen, dann
383  t0_temperaturmessung = millis(); // Timer auf neuen Wert setzen
384  update_messwerte(); // Messwerte aktualisieren
385  }
386 
387 
388  /******************************** Heizung <--> Temperatursensor DHT22 ***********************************/
389  if ( ( millis() - t0_heizung ) > dt_heizung_ms ) { // wenn Zeit für Heizen abgelaufen, dann
390  t0_heizung = millis(); // Timer auf neuen Wert setzen
391  if (temp_luft_C < limit_temp_adc ) { // wenn Lufttemperatur kälter als eingesteller Wert, dann
392  digitalWrite(RELAIS_HEIZUNG, LOW ); // heizen
393  }
394  else { // sonst
395  digitalWrite(RELAIS_HEIZUNG, HIGH ); // nicht heízen
396  }
397  }
398 
399  /******************************** Lüfter <--> Temperatursensor DHT22 ************************************/
400  if ( ( millis() - t0_luefter ) > dt_luefter_ms ) {
401  t0_luefter = millis();
402  if (temp_luft_C > limit_luefter_adc ) { // wenn Lufttemperatur wärmer als eingesteller Wert, dann
403  digitalWrite(RELAIS_LUEFTER, LOW ); // Lüfter einschalten (LOW = Relais EIN)
404  }
405  else { // sonst
406  digitalWrite(RELAIS_LUEFTER, HIGH ); // Lüfter ausschalten (High = Relais AUS)
407  }
408  }
409 
410  /******************************** LED-Licht <--> Helligkeitssensor *************************************/
411  if ( ( millis() - t0_helligkeit ) > dt_helligkeit_ms ) {
412  t0_helligkeit = millis();
413  if ( digitalRead( HELLIGKEITSSENSOR ) == HIGH ) { // wenn dunkel, dann:
414  digitalWrite(RELAIS_UV_LED, LOW ); // Licht anschalten (LOW = Relais EIN)
415  }
416  else { // sonst:
417  digitalWrite(RELAIS_UV_LED, HIGH ); // Licht ausschalten (High = Relais AUS)
418  }
419  }
420 
421 
422  /******************************** Erdfuchtesensor <--> pumpem oben + unten ***************************************/
423  /*
424 
425 
426  // --- Übergangsautomat Erdfeuchte:
427  1. (warten)
428  warten auf "zu trocken", d.h. bis Feuchtigkeit zu niedrig.
429  Wenn zu trocken; Pumpe ein, Statzeit für Pumpe notieren
430  Zustand ändern in -> pumpen
431  2. (pumpen)
432  pumpen, bis Pumpzeit abgelaufen
433  Pumpe aus, Startzeit für "warten auf Durchfeuchtung" notieren,
434  Zustand ändern in -> "warten auf Durchfeuchtung" ändern
435  3. (warten auf durchfeuchtung)
436  wenn Zeit für "warten auf Durchfeuchtung" abgelaufen -> warten auf zu trocken
437  */
438 
439  // für die Pumpe unten
440  switch ( zustand_1 ) {
441 
442  case zustand_warten_zu_trocken :
443  //--- im Zustand bleiben, bis if() erfüllt -> neuer ZUstand
444  if (( hum_erde_1_adc > limit_feuchte_1_adc ) and ( tuer_zu() )) { // wenn Erdfeuchte1 sehr trocken und Tür zu, dann
445  digitalWrite(RELAIS_PUMPE_UNTEN, LOW ); // Pumpe an und wässern (LOW = Relais EIN)
446  t0_pumpe_1 = millis(); // neue Zeit setzen
447  zustand_1 = zustand_pumpt; // neuen Zustand setzen
448  }
449  break;
450 
451  case zustand_pumpt:
452  if (( millis() - t0_pumpe_1 > dt_pumpendauer_1_ms ) or ( not tuer_zu() )) { // wenn die Pumpzeit (Pumpe an) abgelaufen ist, dann
453  digitalWrite(RELAIS_PUMPE_UNTEN, HIGH ); // Pumpe ausschalten (HIGH = Relais AUS)
454  t0_durchfeuchtung_1 = millis(); // neue Zeit setzen
455  zustand_1 = zustand_warten_durchfeuchtung; // neuen Zustand setzen
456  }
457  break;
458 
459  case zustand_warten_durchfeuchtung :
460  if (( millis() - t0_durchfeuchtung_1 > dt_warten_durchfeuchtung_1_ms ) or ( not tuer_zu() )) { // wenn die Sickerzeit des Wassers abgelaufen ist, dann
461  zustand_1 = zustand_warten_zu_trocken; // neuen Zustand setzen um wieder zu messen, ob die Erde trocken oder feucht ist
462  }
463  break;
464  }
465 
466  // für die Pumpe oben
467  switch ( zustand_2 ) {
468 
469  case zustand_warten_zu_trocken :
470  //--- im Zustand bleiben, bis if() erfüllt -> neuer Zustand
471  if (( hum_erde_2_adc > limit_feuchte_2_adc) and ( tuer_zu() )) { // wenn Erdfeuchte2 sehr trocken, dann
472  digitalWrite(RELAIS_PUMPE_OBEN, LOW ); // Pumpe an und wässern (LOW = Relais EIN)
473  t0_pumpe_2 = millis(); // neue Zeit setzen
474  zustand_2 = zustand_pumpt; // neuen Zustand setzen
475  }
476  break;
477 
478  case zustand_pumpt:
479  if (( millis() - t0_pumpe_2 > dt_pumpendauer_2_ms ) or ( not tuer_zu() )) { // wenn die Pumpzeit (Pumpe an) abgelaufen ist, dann
480  digitalWrite(RELAIS_PUMPE_OBEN, HIGH ); // Pumpe ausschalten (HIGH = Relais AUS)
481  t0_durchfeuchtung_2 = millis(); // neue Zeit setzen
482  zustand_2 = zustand_warten_durchfeuchtung; // neuen Zustand setzen
483  }
484  break;
485 
486  case zustand_warten_durchfeuchtung :
487  if (( millis() - t0_durchfeuchtung_2 > dt_warten_durchfeuchtung_2_ms ) or ( not tuer_zu() )) { // wenn die Sickerzeit des Wassers abgelaufen ist, dann
488  zustand_2 = zustand_warten_zu_trocken; // neuen Zustand setzen um wieder zu messen, ob die Erde trocken oder feucht ist
489  }
490  break;
491  }
